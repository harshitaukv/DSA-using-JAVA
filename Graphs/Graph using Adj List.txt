import java.util.*;

class Graph {
    int vertices; // number of vertices
    ArrayList<LinkedList<Integer>> adjList; // adjacency list

    // Constructor
    Graph(int vertices) { // 5
        this.vertices = vertices;
        adjList = new ArrayList<>(vertices);
        // 
        for (int i = 0; i < vertices; i++) {
            adjList.add(new LinkedList<>());
        }
    }

    // Add edge (undirected)
    public void addEdge(int src, int dest) {
        // 0, 1
        adjList.get(src).add(dest);
        adjList.get(dest).add(src);
    }
    
     public void bfs(int start) {
        boolean[] visited = new boolean[vertices];
        System.out.println(Arrays.toString(visited));
        Queue<Integer> queue = new LinkedList<>();

        visited[start] = true;
        queue.add(start);

        System.out.print("BFS starting from " + start + ": ");

        while (!queue.isEmpty()) {
            int vertex = queue.poll();
            System.out.print(vertex + " ");

            for (int neighbor : adjList.get(vertex)) {
                if (!visited[neighbor]) {
                    visited[neighbor] = true;
                    queue.add(neighbor);
                }
            }
        }
        System.out.println();
    }

     public void dfs(int start) {
        boolean[] visited = new boolean[vertices];
        Stack<Integer> stack = new Stack<>();

        stack.push(start);
        System.out.print("DFS starting from " + start + ": ");
        while (!stack.isEmpty()) {
            int vertex = stack.pop();

            if (!visited[vertex]) {
                visited[vertex] = true;
                System.out.print(vertex + " ");

                // Push neighbors in reverse order so smallest appears first
                List<Integer> neighbors = adjList.get(vertex);
                Collections.reverse(neighbors);
                System.out.println("neighbors "+neighbors);
                for (int neighbor : neighbors) {
                    if (!visited[neighbor]) {
                        stack.push(neighbor);
                    }
                }
                Collections.reverse(neighbors); // restore original order
            }
        }
     }
    // Display the graph
    public void printGraph() {
        for (int i = 0; i < vertices; i++) {
            System.out.print(i + " -> ");
            for (int neighbor : adjList.get(i)) {
                System.out.print(neighbor + " ");
            }
            System.out.println();
        }
    }

    // Main method to test
    public static void main(String[] args) {
        Graph g = new Graph(5);

        g.addEdge(0, 1);
        g.addEdge(0, 4);
        g.addEdge(1, 2);
        g.addEdge(1, 3);
        g.addEdge(1, 4);
        g.addEdge(2, 3);
        g.addEdge(3, 4);

        g.printGraph();
        // g.bfs(0);
        // g.dfs(0);
    }
}
